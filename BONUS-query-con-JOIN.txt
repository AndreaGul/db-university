1. Selezionare tutti i corsi del Corso di Laurea in Informatica (22)

SELECT
    `courses`.`name`,
    `courses`.`cfu`,
    `degrees`.`name`
FROM
    `courses`
INNER JOIN `degrees` ON `courses`.`degree_id` = `degrees`.`id`
WHERE
    `degrees`.`name` = "Corso di Laurea in Informatica";
